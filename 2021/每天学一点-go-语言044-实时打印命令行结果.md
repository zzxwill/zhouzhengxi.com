```go

// RealtimePrintCommandOutput prints command output in real time
func RealtimePrintCommandOutput(cmd *exec.Cmd) error {
  writer := io.MultiWriter(os.Stdout)
  cmd.Stdout = writer
  cmd.Stderr = writer
  if err := cmd.Run(); err != nil {
     return err
  }
  return nil
}
```
